FROM                                Contains input graphs
()                                  Contains output structure
WHERE                               Starts conditions
    {}                              Contains conditions
    .                               New condition
    ?                               Use value as variable after it
    =
    <
    >

    in                              Explicit reference to input graph
    <=
    >=
    &&
    ||
    !
    //                              Comments

    UNION                           Combine two graphs
    GROUP
    MAX
    MIN
