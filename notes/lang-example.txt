?g = init "input.ttl" "normInput.ttl" . //Normalises input.ttl to normInput.ttl

?out = "output.ttl" .

//Implicit input graph
SELECT (?s ?p ?obj) FROM (?g) TO (?out) WHERE {
    ?p = <http://example.org/ont/hasAge> .
    ?obj >= 21 .
}

//Explicit input graphs and implicit output (writes to first input file)

    ?p in ?g1 = <http://example.org/ont/hasAge> .
    ?obj >= 21 .
}
